const words = [

    "the", "of", "and", "a", "to", "in", "is", "you", "that", "it", 

    "he", "for", "was", "on", "with", "as", "I", "his", "they", "be", 

    "at", "one", "have", "this", "from", "by", "hot", "but", "some", 

    "what", "there", "we", "can", "out", "other", "were", "all", "your",

    "when", "how", "said", "an", "each", "she", "which", "do", "their",

    "time", "if", "will", "way", "up", "other", "about", "many", "then",

    "them", "write", "would", "like", "so", "these", "her", "long", 

    "make", "things", "see", "him", "two", "has", "look", "more", "go",

    "see", "could", "come", "did", "number", "sound", "no", "most",

    "people", "my", "over", "know", "water", "than", "call", "first", 

    "who", "may", "down", "side", "been", "now", "find", "head", "stand",

    "own", "page", "should", "country", "found", "answer", "school", 

    "grow", "study", "still", "learn", "plant", "cover", "food", "sun"

];


let currentWordIndex = 0;

let startX = null;

const swipeThreshold = 50;


const currentWordElement = document.getElementById('current-word');

const nextButton = document.getElementById('next-button');


function updateWordDisplay() {

    currentWordElement.textContent = words[currentWordIndex];

}


function showNextWord() {

    currentWordIndex = (currentWordIndex + 1) % words.length;

    updateWordDisplay();

}


function showPreviousWord() {

    currentWordIndex = (currentWordIndex - 1 + words.length) % words.length;

    updateWordDisplay();

}


function handleTouchStart(e) {

    startX = e.touches[0].clientX;

}


function handleTouchMove(e) {

    if (!startX) return;


    const currentX = e.touches[0].clientX;

    const diffX = startX - currentX;


    if (Math.abs(diffX) > swipeThreshold) {

        if (diffX > 0) {

            // Left swipe

            showNextWord();

        } else {

            // Right swipe

            showPreviousWord();

        }

        startX = null; // Reset startX after a successful swipe

        e.preventDefault(); // Prevents page from scrolling

    }

}


function handleTouchEnd() {

    startX = null;

}


// Event Listeners

nextButton.addEventListener('click', showNextWord);

currentWordElement.addEventListener('touchstart', handleTouchStart);

currentWordElement.addEventListener('touchmove', handleTouchMove);

currentWordElement.addEventListener('touchend', handleTouchEnd);

currentWordElement.addEventListener('touchcancel', handleTouchEnd);


// Initial word display

updateWordDisplay();
